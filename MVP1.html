      margin: 0;
    }
    canvas {
      display: block;
    }
  </style>
  <script src="https://cdn.jsdelivr.net/npm/three@latest/build/three.min.js"></script>
  <script src="https://cdn.jsdelivr.net/npm/three-obj-mtl-loader@1.0.0/MTLLoader.js"></script>
  <script src="https://cdn.jsdelivr.net/npm/three-obj-mtl-loader@1.0.0/OBJLoader.js"></script>
  <script src="https://cdn.jsdelivr.net/npm/three@latest/examples/js/controls/OrbitControls.js"></script>
</head>
<body>
  <script>
    // Create a scene, camera, and renderer
    const scene = new THREE.Scene();
    const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
    const renderer = new THREE.WebGLRenderer();
    renderer.setSize(window.innerWidth, window.innerHeight);
    document.body.appendChild(renderer.domElement);

    // Add orbit controls
    const controls = new THREE.OrbitControls(camera, renderer.domElement);
    controls.enableDamping = true;
    controls.dampingFactor = 0.05;
    controls.screenSpacePanning = false;
    controls.minDistance = 10;
    controls.maxDistance = 200;

    // Load textures and materials
    const mtlLoader = new THREE.MTLLoader();
    mtlLoader.setPath('https://storage.googleapis.com/drone-mvp-1/');
    mtlLoader.setResourcePath('https://storage.googleapis.com/drone-mvp-1/');
    mtlLoader.load('MVP1.2%20(Holes%20closed).mtl', materials => {
      materials.preload();

      // Load the obj model
      const objLoader = new THREE.OBJLoader();
      objLoader.setMaterials(materials);
      objLoader.setPath('https://storage.googleapis.com/drone-mvp-1/');
      objLoader.load('MVP1.2%20(Holes%20closed).obj', object => {
        scene.add(object);
        object.position.y = -1;
      });
    });

    // Add ambient light
    const ambientLight = new THREE.AmbientLight(0x404040, 1.5);
    scene.add(ambientLight);

    // Add directional light
    const directionalLight = new THREE.DirectionalLight(0xffffff, 1);
    directionalLight.position.set(1, 1, 1).normalize();
    scene.add(directionalLight);

    camera.position.z = 50;

    // Animation loop
    const animate = function () {
      requestAnimationFrame(animate);
      controls.update();
      renderer.render(scene, camera);
    };

    animate();
  </script>
</body>
